Sales Performance & Trend Analysis

Insights from 40,000+ sales records

Python Code


import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load dataset
df = pd.read_csv("sales_data.csv", encoding='latin1', engine='python')
df.columns = df.columns.str.strip()

# Convert date safely
df['Order Date'] = pd.to_datetime(df['Order Date'], errors='coerce')

# Drop missing values
df = df.dropna(subset=['Order Date', 'Sales'])

# Feature Engineering
df['Year'] = df['Order Date'].dt.year
df['Month'] = df['Order Date'].dt.month
df['Quarter_num'] = df['Order Date'].dt.quarter

# Total Sales
print("Total Sales:", df['Sales'].sum())

# Sales by Quarter
quarter_sales = df.groupby('Quarter_num')['Sales'].sum()
print("\nSales by Quarter:\n", quarter_sales)

# Q4 Analysis
q4_sales = quarter_sales.get(4, 0)
other_quarters = quarter_sales.drop(4, errors='ignore')
if len(other_quarters) > 0:
    avg_other = other_quarters.mean()
    increase = ((q4_sales - avg_other) / avg_other) * 100
    print(f"\nQ4 Revenue Increase: {increase:.2f}%")

# Monthly Trend
monthly_sales = df.groupby('Month')['Sales'].sum()
plt.figure()
monthly_sales.plot(marker='o')
plt.show()

# Category Sales
plt.figure()
sns.barplot(data=df, x='Category', y='Sales')
plt.show()
